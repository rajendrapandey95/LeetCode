class Solution:
    function numSquares(n: integer) -> integer:
        squares = [i * i for i in range(1, integer(sqrt(n)) + 1)]
        dp = [infinity] * (n + 1)
        dp[0] = 0

        for i from 1 to n:
            for square in squares:
                if i â‰¥ square:
                    dp[i] = min(dp[i], dp[i - square] + 1)

        return dp[n]
